# Ch05. 함수 - 2



## OUT

	- ref키워드처럼 레퍼런스를 가진다

 - 전달하는 변수의 초기화 없이 사용 가능하다
   	- static void InitNum(out int addNum)



#### OUT 예제

```C#
namespace _063_Func_out
{
    class Program
    {
        static void InitNum(out int addNum)
        {
            addNum = 100;
        }

        static void InitRefNum(ref int refNum)
        {
            refNum = 100;
        }

        static void Main(string[] args)
        {
            int a;
            int b;

            InitNum(out a); //a에 100을 대입하는 함수
            Console.WriteLine("a: " + a);

            b = 0;
            InitRefNum(ref b); //오류: b를 초기화 하지 않고 사용 불가!! 즉 b를 0으로 초기화한 뒤 사용해야 한다.
            Console.WriteLine("b: " + b);
        }
    }
}
```



## 디폴트 매개 변수(Default parameter)

	- 초기화 파라미터
	- 반드시 파라미터 뒤에서부터 순차적으로 초기화한다.



```C#
namespace _064_Func_default
{
    class Program
    {
        static void PrintValue(int a, int b, int c = 100, int d = 0) // c와 d에 특정 수를 대입하여 default값을 설정한다.
            //중요한 point는 뒤쪽의 파라미터부터 순차적으로 대입하여야 한다. c를 default설정하지 않고 b를 default값 설정 시 오류 발생.
        {
            Console.WriteLine("PrintValue:  {0}  {1}  {2}  {3}",a , b, c, d);
        }

        static void Main(string[] args)
        {
            PrintValue(0, 0, 0, 0);
            PrintValue(100, 2, 1);
            PrintValue(300, 300);
        }
    }
}
/* 결과 창 */
PrintValue:  0  0  0  0
PrintValue:  100  2  1  0 //앞에서부터 차례대로 100, 2, 1이 대입되었다. c: 100에서 1로 변화
PrintValue:  300  300  100  0
```



## 메서드 오버로딩

 - 메서드 이름이 중복된다
    - 대신 파라미터 형식을 다르게 하거나,
    - 파라미터 의 수를 다르게 해야 한다.
      	- static int Add(int a, int b)
      	- static int Add(int a, int b, int c)
      	- static int Add(float a, float b)



```c#
namespace _065_Func_Overloading
{
    class Program
    {
        static int Add(int a, int b)
        {
            return a + b;
        }

        static int Add(int a, int b, int c)
        {
            return a + b + c;
        }

        static double Add(double a, double b)
        {
            return a + b;

        }

        //Add(int a, int b, int c) 모호성으로 오류 발생
        static int Add(int a, int b, int c, int d = 0) //d를 선언하지 않을 경우 위의 2번째 함수가 호출될 수 있기 때문에 모호성이 발생한다.
        {
            return (a + b + c + d) * 2;
        }

        static void Main(string[] args)
        {
            Console.WriteLine("Add(int a, int b): " + Add(10, 100));
            Console.WriteLine("Add(int a, int b, int c): " + Add(10, 100, 1000));
            Console.WriteLine("Add(double a, double b): " + Add(10.3, 100.5));
        }
    }
}
/* 결과 창 */
Add(int a, int b): 110
Add(int a, int b, int c): 1110
Add(double a, double b): 110.8
```



## params



 - 파라미터의 수에 제한 받지 않고 처리할 수 있다.
   	- int Total(params int[] values) : int값을 원하는 만큼 파라미터로 전달받을 수 있다.



```C#
namespace _066_Func_params
{
    class Program
    {
        static int Total(params int[] values) // 내가 params로 전달하는 여러개의 int형 변수를 values에 차례로 전달한다.
        {
            int total = 0; 

            for (int i = 0; i < values.Length; i++) //i를 0부터 n-1까지의 index까지 증가시키면서
            {
                total += values[i]; //total값에 합산한다.
            }
            return total;
        }

        static void CheckValues(params object[] values)
		//object는 모든 형태의 변수를 처리할 수 있으나, boxing/unboxing이 일어나며 메모리 상에서의 손해가 많기 때문에 추천하지 않는다.
        {
            for (int i = 0; i < values.Length; i++)
            {
                Console.WriteLine("Value: {0}   {1}", values[i], values[i].GetType() );
            }
        }

        static void Main(string[] args)
        {
            Console.WriteLine("Total: " + Total(200, 5, 3, 1, 100));
            Console.WriteLine("Total: " + Total(1, 2, 3, 4, 5));
            Console.WriteLine("Total: " + Total(10, 1000));

            CheckValues(0, 10.2f, 10.4, 10434839840d, "Hello World!");
        }
    }
}
/* 결과 창 */
Total: 309
Total: 15
Total: 1010
Value: 0   System.Int32
Value: 10.2   System.Single
Value: 10.4   System.Double
Value: 10434839840   System.Double
Value: Hello World!   System.String
```
